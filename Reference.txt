def get_products():
    if connection:
        cursor = connection.cursor()
        try:
            cursor.execute(
                """
                SELECT * FROM PRODUCTS
                """
            )
            print("Data retrieved successfully")
            return cursor.fetchall()

            print("Error fetching data:", e)
            return None

@app.route("/")
